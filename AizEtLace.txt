    <script>
        let currentUser = null;
        let cart = [];
        let products = {
            aesthetic: [],
            coquete: [],
            vintage: []
        };

        function showCatalog(category) {
            document.querySelectorAll('.catalog').forEach(cat => cat.classList.remove('active'));
            document.getElementById(category + '-catalog').classList.add('active');
            renderProducts(category);
        }

        function renderProducts(category) {
            const container = document.getElementById(category + '-products');
            container.innerHTML = '';
            products[category].forEach(product => {
                const productDiv = document.createElement('div');
                productDiv.className = 'product';
                productDiv.innerHTML = `
                    <img src="${product.image}" alt="${product.name}">
                    <h4>${product.name}</h4>
                    <p>Rp ${product.price}</p>
                    <p>${product.description}</p>
                    <p>Toko: ${product.store}</p>
                    <button onclick="addToCart('${product.name}', ${product.price})">Tambah ke Keranjang</button>
                `;
                container.appendChild(productDiv);
            });
        }

        function addToCart(name, price) {
            cart.push({ name, price });
            updateCartCount();
        }

        function updateCartCount() {
            document.getElementById('cart-count').textContent = cart.length;
        }

        function openCart() {
            const cartItems = document.getElementById('cart-items');
            cartItems.innerHTML = '';
            cart.forEach(item => {
                const itemDiv = document.createElement('div');
                itemDiv.textContent = `${item.name} - Rp ${item.price}`;
                cartItems.appendChild(itemDiv);
            });
            document.getElementById('cart-modal').style.display = 'flex';
        }

        function checkout() {
            const message = cart.map(item => `${item.name} - Rp ${item.price}`).join('\n');
            const whatsappUrl = `https://wa.me/6287862438374?text=${encodeURIComponent('Checkout:\n' + message)}`;
            window.location.href = whatsappUrl;
        }

        function openModal(type) {
            document.getElementById('auth-modal').style.display = 'flex';
            if (type === 'login') {
                document.getElementById('login-form').style.display = 'block';
                document.getElementById('signup-form').style.display = 'none';
            } else {
                document.getElementById('login-form').style.display = 'none';
                document.getElementById('signup-form').style.display = 'block';
            }
        }

        function closeModal() {
            document.querySelectorAll('.modal').forEach(modal => modal.style.display = 'none');
        }

        function toggleForm() {
            const login = document.getElementById('login-form');
            const signup = document.getElementById('signup-form');
            if (login.style.display === 'none') {
                login.style.display = 'block';
                signup.style.display = 'none';
            } else {
                login.style.display = 'none';
                signup.style.display = 'block';
            }
        }

        function login(event) {
            event.preventDefault();
            // Simulasi login
            currentUser = { email: event.target[0].value };
            alert('Login berhasil!');
            closeModal();
            // Tambahkan tombol upload produk di header atau navigasi
            const nav = document.querySelector('nav');
            const uploadBtn = document.createElement('a');
            uploadBtn.href = '#';
            uploadBtn.textContent = 'Upload Produk';
            uploadBtn.onclick = () => document.getElementById('upload-modal').style.display = 'flex';
            nav.appendChild(uploadBtn);
        }

        function signup(event) {
            event.preventDefault();
            // Simulasi signup
            currentUser = { store: event.target[0].value, email: event.target[1].value };
            alert('Sign up berhasil! Sekarang Anda bisa membuat toko.');
            closeModal();
            // Tambahkan tombol upload
            const nav = document.querySelector('nav');
            const uploadBtn = document.createElement('a');
            uploadBtn.href = '#';
            uploadBtn.textContent = 'Upload Produk';
            uploadBtn.onclick = () => document.getElementById('upload-modal').style.display = 'flex';
            nav.appendChild(uploadBtn);
        }

        function uploadProduct(event) {
            event.preventDefault();
            const category = event.target[0].value;
            const name = event.target[1].value;
            const price = event.target[2].value;
            const description = event.target[3].value;
            const file = event.target[4].files[0];
            const reader = new FileReader();
            reader.onload = function(e) {
                const product = {
                    name,
                    price,
                    description,
                    image: e.target.result,
                    store: currentUser.store || 'Toko Anda'
                };
                products[category].push(product);
                renderProducts(category);
                closeModal();
            };
            reader.readAsDataURL(file);
        }
    </script>
